BillDownloader/
├── src/
│   ├── main/                  # 主进程（Node.js/Electron API）
│   │   ├── coreServices/      # 核心服务层实现
│   │   │   ├── networkService.js      # 网络请求服务（封装axios）
│   │   │   ├── parserService.js       # 数据解析服务（cheerio/m3u8-parser）
│   │   │   ├── mediaService.js        # 媒体处理服务（fluent-ffmpeg）
│   │   │   ├── taskService.js         # 任务管理服务（队列/状态同步）
│   │   │   ├── storageService.js      # 数据存储服务（sqlite3/electron-store）
│   │   │   └── securityService.js     # 权限与风控服务（Cookie/设备指纹）
│   │   ├── business/          # 业务功能层实现
│   │   │   ├── videoDownload/         # 视频下载子系统
│   │   │   ├── articleCrawl/          # 专栏爬取子系统
│   │   │   └── liveRecord/            # 直播录制子系统
│   │   ├── utils/             # 工具函数（日志/系统交互）
│   │   └── main.js            # 主进程入口（窗口管理/IPC注册）
│   ├── renderer/              # 渲染进程（React/Fluent UI）
│   │   ├── components/        # UI组件（导航/任务列表/设置面板）
│   │   ├── pages/             # 页面（视频下载页/专栏页/直播页）
│   │   ├── store/             # 状态管理（任务状态/用户设置）
│   │   ├── ipc/               # IPC通信封装（渲染进程→主进程）
│   │   └── main.jsx           # 渲染进程入口（React挂载）
│   └── assets/                # 静态资源（图标/样式）
├── database/                  # sqlite3数据库文件
├── config/                    # 配置文件（默认设置/更新地址）
└── scripts/                   # 打包/部署脚本（electron-builder）
